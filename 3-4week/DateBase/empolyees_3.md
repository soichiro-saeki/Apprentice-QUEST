・employeesテーブルをeに命名  

SELECT d.dept_no, dm.emp_no, e.first_name, e.last_name
FROM departments d 　　　　
JOIN dept_manager dm　　　
ON d.dept_no = dm.dept_no
JOIN employees e 
ON dm.emp_no = e.emp_no
ORDER BY d.dept_no;
